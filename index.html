class User extends Authenticatable implements MustVerifyEmail
{
    use HasApiTokens, HasFactory, Notifiable, SoftDeletes;

    protected $fillable = [
        'name','email','phone','password','role',
        'avatar','locale','currency','is_active',
        'two_factor_enabled','two_factor_secret',
    ];
    protected $hidden = ['password','remember_token','two_factor_secret'];
    protected $casts  = [
        'email_verified_at'    => 'datetime',
        'is_active'            => 'boolean',
        'two_factor_enabled'   => 'boolean',
        'password'             => 'hashed',
    ];

    public function orders()          { return $this->hasMany(Order::class); }
    public function cart()            { return $this->hasOne(Cart::class); }
    public function addresses()       { return $this->hasMany(Address::class); }
    public function reviews()         { return $this->hasMany(Review::class); }
    public function wishlist()        { return $this->hasMany(Wishlist::class); }
    public function products()        { return $this->hasMany(Product::class); }
    public function loyaltyPoints()   { return $this->hasMany(LoyaltyPoint::class); }
    public function recentlyViewed()  { return $this->hasMany(RecentlyViewed::class)->latest('viewed_at'); }

    public function isAdmin(): bool   { return $this->role === 'admin'; }
    public function isVendor(): bool  { return $this->role === 'vendor'; }

    public function getTotalLoyaltyPointsAttribute(): int
    {
        return $this->loyaltyPoints()->sum('points');
    }

    public function defaultAddress(): ?Address
    {
        return $this->addresses()->where('is_default', true)->first();
    }
}

// ════════════════════════════════════════════════
class Coupon extends \Illuminate\Database\Eloquent\Model
{
    use HasFactory;
    protected $fillable = [
        'code','name','type','value','min_order_amount','max_discount',
        'max_uses','max_uses_per_user','used_count','is_active','starts_at','expires_at',
    ];
    protected $casts = [
        'value'            => 'decimal:2',
        'min_order_amount' => 'decimal:2',
        'max_discount'     => 'decimal:2',
        'is_active'        => 'boolean',
        'starts_at'        => 'datetime',
        'expires_at'       => 'datetime',
    ];

    public function isValid(float $amount): bool
    {
        if (!$this->is_active) return false;
        if ($this->starts_at  && $this->starts_at->isFuture())  return false;
        if ($this->expires_at && $this->expires_at->isPast())   return false;
        if ($this->max_uses   && $this->used_count >= $this->max_uses) return false;
        if ($amount < $this->min_order_amount) return false;
        return true;
    }

    public function calculateDiscount(float $amount): float
    {
        $discount = match ($this->type) {
            'percent'      => round($amount * ($this->value / 100), 2),
            'fixed'        => min((float)$this->value, $amount),
            'free_shipping'=> 0,
            default        => 0,
        };
        if ($this->max_discount) {
            $discount = min($discount, (float)$this->max_discount);
        }
        return $discount;
    }
}

            'name'     => $request->string('name')->trim()->value(),
            'email'    => $request->string('email')->lower()->value(),
            'phone'    => $request->input('phone'),
            'password' => Hash::make($request->input('password')),

        $user->sendEmailVerificationNotification();
        event(new \App\Events\UserRegistered($user));

        $token = $user->createToken('api')->plainTextToken;

            'user'  => new \App\Http\Resources\UserResource($user),
            'token' => $token,
        ], 201);

    {
        if (!Auth::attempt($request->only('email', 'password'))) {
            return response()->json(['message' => 'Invalid credentials.'], 401);
        }

        $user = auth()->user();
        abort_if(!$user->is_active, 403, 'Account suspended.');

        // Merge guest cart
        if ($sessionId = $request->header('X-Session-Id')) {
            app(CartService::class)->mergeGuestCart($sessionId, $user->id);
        }

        $token = $user->createToken('api')->plainTextToken;

        return response()->json([
            'user'  => new \App\Http\Resources\UserResource($user),
            'token' => $token,
        ]);
    }

    public function logout(Request $request)
    {
        $request->user()->currentAccessToken()->delete();
        return response()->json(['message' => 'Logged out.']);
    }

    {
        return new \App\Http\Resources\UserResource($request->user()->load('addresses'));
    }

            'reviews' => fn($q) => $q->where('is_approved',true)->with('user')->latest()->limit(10)
        ]);

            \App\Models\RecentlyViewed::updateOrCreate(
                ['user_id' => auth()->id(), 'product_id' => $product->id],
                ['viewed_at' => now()]
            );

    {
        $payload   = $request->all();
        $signature = $request->header('Stripe-Signature');

        try {
            \Stripe\Webhook::constructEvent(
                $request->getContent(),
                $signature,
                config('services.stripe.webhook_secret')
            );
        } catch (\Exception $e) {
            return response()->json(['error' => 'Invalid signature.'], 400);
        }

        $this->paymentService->handleStripeWebhook($payload);
        return response()->json(['received' => true]);
    }
}

class CartService
{
    public function getOrCreate(?int $userId, ?string $sessionId): \App\Models\Cart
    {
        if ($userId) return \App\Models\Cart::firstOrCreate(['user_id' => $userId]);
        return \App\Models\Cart::firstOrCreate(['session_id' => $sessionId]);
    }

    public function add(\App\Models\Cart $cart, int $productId, int $quantity, ?int $variantId = null): void
    {
        $product = \App\Models\Product::findOrFail($productId);
        abort_if($product->status !== 'active', 422, 'Product unavailable.');

        if ($variantId) {
            $variant = \App\Models\ProductVariant::findOrFail($variantId);
            abort_if($variant->stock < $quantity, 422, 'Not enough stock for this variant.');
            $price = $variant->current_price;
        } else {
            abort_if($product->stock < $quantity, 422, 'Not enough stock.');
            $price = $product->current_price;
        }

        if ($quantity === 0) {
            $cart->items()->where('product_id', $productId)
                ->where('product_variant_id', $variantId)->delete();
            return;
        }

        $cart->items()->updateOrCreate(
            ['product_id' => $productId, 'product_variant_id' => $variantId],
            ['quantity' => $quantity, 'price' => $price]
        );
    }

    public function clear(int $userId): void
    {
        \App\Models\Cart::where('user_id', $userId)->first()?->items()->delete();
    }

    public function mergeGuestCart(string $sessionId, int $userId): void
    {
        $guest = \App\Models\Cart::where('session_id', $sessionId)->with('items')->first();
        if (!$guest || $guest->items->isEmpty()) return;

        $userCart = $this->getOrCreate($userId, null);
        foreach ($guest->items as $item) {
            $userCart->items()->updateOrCreate(
                ['product_id' => $item->product_id, 'product_variant_id' => $item->product_variant_id],
                ['quantity' => $item->quantity, 'price' => $item->price]
            );
        }
        $guest->items()->delete();
        $guest->delete();
    }
}

        abort_if($cart->items->isEmpty(), 422, 'Cart is empty.');

        return \DB::transaction(function () use ($data, $userId, $cart) {
            $subtotal = $cart->total;
            $discount = 0;
            $coupon   = null;

            if (!empty($data['coupon'])) {

            $taxRate     = config('shop.tax_rate', 0);
            $tax         = round(($subtotal - $discount) * $taxRate, 2);

                'coupon_id'        => $coupon?->id,

                'discount'         => $discount,
                'shipping_fee'     => $shippingFee,
                'tax'              => $tax,

                'shipping_country' => $data['shipping_country'] ?? 'KH',
                'note'             => $data['note'] ?? null,
                'is_gift'          => $data['is_gift'] ?? false,
                'gift_message'     => $data['gift_message'] ?? null,
                'ip_address'       => $data['ip'] ?? null,
            ]);

            foreach ($cart->items as $item) {
                $stockField = $item->variant_id ? 'variant' : 'product';
                $stock = $item->variant ? $item->variant->stock : $item->product->stock;
                abort_if($stock < $item->quantity, 422, "Insufficient stock: {$item->product->name}");

                $order->items()->create([
                    'product_id'         => $item->product_id,
                    'product_variant_id' => $item->product_variant_id,

                    'total_price'        => $item->price * $item->quantity,

                } else {

        });
    }

                'delivered' => ['delivered_at' => now()],
                'cancelled' => ['cancelled_at' => now()],
                default     => [],
            };

            $order->histories()->create([
                'user_id'          => $adminId,
                'status'           => $status,
                'note'             => $note,
                'notify_customer'  => $notifyCustomer,
            ]);

            if ($status === 'cancelled') {
                foreach ($order->items as $item) {
                    if ($item->product_variant_id) {
                        $item->variant->increment('stock', $item->quantity);
                    } else {
                        $item->product->increment('stock', $item->quantity);
                    }
                }
            }
            event(new \App\Events\OrderStatusChanged($order, $status));
        });
    }

class CouponService
{
    public function validate(string $code, float $amount, int $userId): \App\Models\Coupon
    {
        $coupon = \App\Models\Coupon::where('code', strtoupper($code))->firstOrFail();
        abort_if(!$coupon->isValid($amount), 422, 'Coupon is invalid, expired, or minimum amount not met.');

        $userUsage = \App\Models\CouponUser::where('coupon_id', $coupon->id)->where('user_id', $userId)->value('used_count') ?? 0;
        abort_if($userUsage >= $coupon->max_uses_per_user, 422, 'You have already used this coupon.');

        return $coupon;
    }

    public function apply(\App\Models\Coupon $coupon, float $amount, int $userId): float
    {
        $discount = $coupon->calculateDiscount($amount);
        $coupon->increment('used_count');
        \App\Models\CouponUser::updateOrCreate(
            ['coupon_id' => $coupon->id, 'user_id' => $userId],
            ['used_count' => \DB::raw('used_count + 1')]
        );
        return $discount;
    }
}

            'currency' => strtolower($order->currency),

        ]);
        $order->update(['payment_reference' => $intent->id]);
        return ['client_secret' => $intent->client_secret, 'payment_intent_id' => $intent->id];
    }

    public function handleStripeWebhook(array $payload): void
    {
        match ($payload['type'] ?? '') {
            'payment_intent.succeeded'      => $this->markPaid($payload['data']['object']),
            'payment_intent.payment_failed' => $this->markFailed($payload['data']['object']),
            default => null,
        };
    }

    private function markPaid(array $intent): void
    {
        $order = \App\Models\Order::where('payment_reference', $intent['id'])->first();
        if (!$order) return;
        $order->update(['payment_status' => 'paid', 'paid_at' => now(), 'status' => 'confirmed']);
        event(new \App\Events\OrderStatusChanged($order, 'confirmed'));
    }

    private function markFailed(array $intent): void
    {
        \App\Models\Order::where('payment_reference', $intent['id'])->update(['payment_status' => 'failed']);
    }
}

namespace App\Jobs;

use Illuminate\Bus\Queueable;

use Illuminate\Queue\{InteractsWithQueue, SerializesModels};

class SendOrderConfirmationEmail implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;
    public int $tries = 3;
    public function __construct(public \App\Models\Order $order) {}
    public function handle(): void
    {
        $this->order->load('user','items.product','coupon');
        \Mail::to($this->order->user->email)->send(new \App\Mail\OrderConfirmationMail($this->order));
    }
}

class GenerateInvoicePdf implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;
    public function __construct(public \App\Models\Order $order) {}
    public function handle(): void
    {
        $this->order->load('user','items.product','coupon');
        $pdf  = \Barryvdh\DomPDF\Facade\Pdf::loadView('invoices.order', ['order' => $this->order]);
        $path = "invoices/order-{$this->order->order_number}.pdf";
        \Storage::disk('private')->put($path, $pdf->output());
        $this->order->update(['invoice_path' => $path]);
    }
}

    }
}

    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;
    public function __construct(public \App\Models\Product $product) {}
    public function handle(): void
    {
        $this->product->vendor->notify(new \App\Notifications\LowStockAlert($this->product));
    }
}

class SendAbandonedCartReminder implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;
    public function __construct(public \App\Models\Cart $cart) {}
    public function handle(): void
    {
        $cart = $this->cart->load('user','items.product');
        if (!$cart->user || $cart->items->isEmpty()) return;
        if ($cart->updated_at->diffInHours(now()) < 24) return;
        \Mail::to($cart->user->email)->send(new \App\Mail\AbandonedCartMail($cart));
    }
}

namespace App\Observers;

class ProductObserver
{
    public function updated(\App\Models\Product $p): void
    {
        $this->clearCache($p);
        if ($p->isDirty('stock') && $p->stock <= $p->min_stock_alert && $p->stock > 0) {
            \App\Jobs\SendLowStockAlert::dispatch($p)->delay(now()->addMinutes(5));
        }
    }
    public function created(\App\Models\Product $p): void { $this->clearCache($p); }
    public function deleted(\App\Models\Product $p): void { $this->clearCache($p); }
    private function clearCache(\App\Models\Product $p): void
    {
        \Cache::forget("product_{$p->id}");
        \Cache::forget("product_{$p->slug}");
        \Cache::forget('featured_products');
        \Cache::forget('admin_dashboard_stats');
    }
}

{
    public function created(\App\Models\Order $o): void  { \Cache::forget('admin_dashboard_stats'); }
    public function updated(\App\Models\Order $o): void  { \Cache::forget('admin_dashboard_stats'); }

namespace App\Observers;

class ProductObserver
{
    public function updated(\App\Models\Product $p): void
    {
        $this->clearCache($p);
        if ($p->isDirty('stock') && $p->stock <= $p->min_stock_alert && $p->stock > 0) {
            \App\Jobs\SendLowStockAlert::dispatch($p)->delay(now()->addMinutes(5));
        }
    }
    public function created(\App\Models\Product $p): void { $this->clearCache($p); }
    public function deleted(\App\Models\Product $p): void { $this->clearCache($p); }
    private function clearCache(\App\Models\Product $p): void
    {
        \Cache::forget("product_{$p->id}");
        \Cache::forget("product_{$p->slug}");
        \Cache::forget('featured_products');
        \Cache::forget('admin_dashboard_stats');
    }
}

class OrderObserver
{
    public function created(\App\Models\Order $o): void  { \Cache::forget('admin_dashboard_stats'); }
    public function updated(\App\Models\Order $o): void  { \Cache::forget('admin_dashboard_stats'); }
}

}

namespace App\Observers;

class ProductObserver
{
    public function updated(\App\Models\Product $p): void
    {
        $this->clearCache($p);
        if ($p->isDirty('stock') && $p->stock <= $p->min_stock_alert && $p->stock > 0) {
            \App\Jobs\SendLowStockAlert::dispatch($p)->delay(now()->addMinutes(5));
        }
    }
    public function created(\App\Models\Product $p): void { $this->clearCache($p); }
    public function deleted(\App\Models\Product $p): void { $this->clearCache($p); }
    private function clearCache(\App\Models\Product $p): void
    {
        \Cache::forget("product_{$p->id}");
        \Cache::forget("product_{$p->slug}");
        \Cache::forget('featured_products');
        \Cache::forget('admin_dashboard_stats');
    }
}

class OrderObserver
{
    public function created(\App\Models\Order $o): void  { \Cache::forget('admin_dashboard_stats'); }
    public function updated(\App\Models\Order $o): void  { \Cache::forget('admin_dashboard_stats'); }
}

class OrderPolicy
{
    public function view(\App\Models\User $user, \App\Models\Order $order): bool   { return $user->isAdmin() || $order->user_id === $user->id; }

    protected $description = 'Send abandoned cart reminder emails';

            ->where('updated_at','>',now()->subHours(72))
            ->whereHas('items')
            ->each(fn($cart) => \App\Jobs\SendAbandonedCartReminder::dispatch($cart));
        $this->info('Reminders queued.');
    }
}

namespace App\Jobs;

class SendOrderConfirmationEmail implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;
    public int $tries = 3;
    public function __construct(public \App\Models\Order $order) {}
    public function handle(): void
    {
        $this->order->load('user','items.product','coupon');
        \Mail::to($this->order->user->email)->send(new \App\Mail\OrderConfirmationMail($this->order));
    }
}

class GenerateInvoicePdf implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;
    public function __construct(public \App\Models\Order $order) {}
    public function handle(): void
    {
        $this->order->load('user','items.product','coupon');
        $pdf  = \Barryvdh\DomPDF\Facade\Pdf::loadView('invoices.order', ['order' => $this->order]);
        $path = "invoices/order-{$this->order->order_number}.pdf";
        \Storage::disk('private')->put($path, $pdf->output());
        $this->order->update(['invoice_path' => $path]);
    }
}

class SendAbandonedCartReminder implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;
    public function __construct(public \App\Models\Cart $cart) {}
    public function handle(): void
    {
        $cart = $this->cart->load('user','items.product');
        if (!$cart->user || $cart->items->isEmpty()) return;
        if ($cart->updated_at->diffInHours(now()) < 24) return;
        \Mail::to($cart->user->email)->send(new \App\Mail\AbandonedCartMail($cart));
    }
}

            'delivered' => new \App\Notifications\OrderDelivered($event->order),
            default     => null,
        };
        if ($notification) $event->order->user->notify($notification);
    }
}

{
    public function handle(\App\Events\UserRegistered $event): void

        \App\Jobs\SendWelcomeEmailJob::dispatch($event->user);

namespace App\Jobs;

use Illuminate\Bus\Queueable;

class SendOrderConfirmationEmail implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;
    public int $tries = 3;
    public function __construct(public \App\Models\Order $order) {}
    public function handle(): void
    {
        $this->order->load('user','items.product','coupon');
        \Mail::to($this->order->user->email)->send(new \App\Mail\OrderConfirmationMail($this->order));
    }
}

class GenerateInvoicePdf implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;
    public function __construct(public \App\Models\Order $order) {}
    public function handle(): void
    {
        $this->order->load('user','items.product','coupon');
        $pdf  = \Barryvdh\DomPDF\Facade\Pdf::loadView('invoices.order', ['order' => $this->order]);
        $path = "invoices/order-{$this->order->order_number}.pdf";
        \Storage::disk('private')->put($path, $pdf->output());
        $this->order->update(['invoice_path' => $path]);
    }
}

class SendAbandonedCartReminder implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;
    public function __construct(public \App\Models\Cart $cart) {}
    public function handle(): void
    {
        $cart = $this->cart->load('user','items.product');

        if ($cart->updated_at->diffInHours(now()) < 24) return;
        \Mail::to($cart->user->email)->send(new \App\Mail\AbandonedCartMail($cart));
    }
}

namespace App\Observers;

class ProductObserver
{
    public function updated(\App\Models\Product $p): void
    {
        $this->clearCache($p);
        if ($p->isDirty('stock') && $p->stock <= $p->min_stock_alert && $p->stock > 0) {
            \App\Jobs\SendLowStockAlert::dispatch($p)->delay(now()->addMinutes(5));
        }
    }
    public function created(\App\Models\Product $p): void { $this->clearCache($p); }
    public function deleted(\App\Models\Product $p): void { $this->clearCache($p); }
    private function clearCache(\App\Models\Product $p): void
    {
        \Cache::forget("product_{$p->id}");
        \Cache::forget("product_{$p->slug}");
        \Cache::forget('featured_products');
        \Cache::forget('admin_dashboard_stats');
    }
}

class OrderObserver
{
    public function created(\App\Models\Order $o): void  { \Cache::forget('admin_dashboard_stats'); }
    public function updated(\App\Models\Order $o): void  { \Cache::forget('admin_dashboard_stats'); }
}

    public function delete(\App\Models\User $user, \App\Models\Product $p): bool  { return $user->isAdmin() || $p->user_id === $user->id; }
}

}
